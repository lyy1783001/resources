import torch

def cosine_topk(query, database, k=3, use_similarity=True):
    """
    计算query与database中每个向量的余弦相似度，并返回每个query在database中相似度最高的k个索引。
    若use_similarity=False，则返回余弦距离（1 - 相似度）最小的k个索引。
    
    Args:
        query: Tensor of shape (m, d) - 查询向量矩阵
        database: Tensor of shape (n, d) - 数据库向量矩阵
        k: int - 返回的top个数
        use_similarity: bool - True返回相似度最大的索引，False返回距离最小的索引
    
    Returns:
        values: Tensor of shape (m, k) - 每个query对应的top-k相似度（或距离）值
        indices: Tensor of shape (m, k) - 每个query对应的top-k索引（在database中的行号）
    """
    # 确保输入为浮点型
    query = query.float()
    database = database.float()
    
    # 计算点积矩阵 (m, n)
    dot = torch.mm(query, database.T)
    
    # 计算L2范数，避免除零
    query_norm = torch.norm(query, dim=1, keepdim=True)  # (m, 1)
    db_norm = torch.norm(database, dim=1, keepdim=True)  # (n, 1)
    
    # 余弦相似度 = 点积 / (||q|| * ||db||)
    # 添加一个小epsilon防止除以零
    similarity = dot / (torch.mm(query_norm, db_norm.T) + 1e-8)
    
    if use_similarity:
        # 取相似度最大的k个
        values, indices = torch.topk(similarity, k, dim=1, largest=True)
    else:
        # 取余弦距离（1 - 相似度）最小的k个
        distance = 1 - similarity
        values, indices = torch.topk(distance, k, dim=1, largest=False)
    
    return values, indices
